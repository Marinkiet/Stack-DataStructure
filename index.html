#Stack Data Structure

Read and learn
  • A Stack is a linear Data Structure which operates in a FILO/LIFO pattern.
  • It behaves like a real world stack of objects. e.g. deck of cards or a stack of books.
  • It allowes adding & removing elements in a particular order[LIFO/FILO].
  • Arrays help us visualize it better.

Here are the standard mathods you'll use for stacks

  1. Push() :This places an item onto the stack. If there's no place for a new item, the stack is considered an Overflow State.
  2. Pop() :This returns the item at the top of the stack and then removes it. If pop is called when the stack is empty, the stack is considered an Underfloew State.
  3. isEmpty() :This determines if the stack is empty or not.
  4. isFull() :This determines if the stack id full or not.
  5. peek() :This lets you view the item at a specific position.
  6. count() :This lets you view the number of items present in the stack.
  7. change() :This lets you change/update an item at a specific position.
  8. display() :This lets you view all the items in the stack.

Technical Examples:

  • Undo functionality in editors/games/apps .On click of the undo button the last task performed is popped of the stack of your project
  • The cache memory in web browsers.Notice how the last visited web page loads faster.

//Copy and run this in your c++ code editor

#include<iostream>
#include<string>
    
    using namespace std;
    
    class Stack{
        private:
            int top;
            int arr[5];
            
            public:
                Stack(){
                    top=-1;
                    
                    for(int i = 0;i <5; i++){
                        arr[i]=0;
                    }
                    
                }
        //methods
            bool isEmpty(){
                if(top==-1)
                    return true;
                else
                    return false;
            }
            
            bool isFull(){
                if(top==4)
                    return true;
                else
                    return false;
            }
            void push(int val){
                if(isFull())
                    cout<<"Stack overflow";
                else{
                    top++;
                    arr[top]=val;
                }
                
            }
            int pop(){
                if(isEmpty()){
                    cout<<"Stack underflow";
                    return 0;
                }
                    
                else{
                    int popVal = arr[top];
                    arr[top]=0;
                    top--;
                    return popVal;
                }
            }	
                
            int count(){
                return (top+1);
            }
            
            int peek(int pos){
                
                if(isEmpty()){
                    cout<<"Stack underflow";
                    return 0;
                }	
                else{
                    return arr[pos];
                }
            }	
                
            void change(int pos,int val){
                arr[pos]=val;
                cout<<"Value changes at location"<<pos<<endl;
            }
            
            void display(){
                cout<<"All the values in the stack: "<<endl;
                for(int i = 4; i >=0;i--){
                    cout<<arr[i]<<" ";
                }
            
            }
    };
    
    
    
    
    int main(){
       
           Stack stk;
           int option,position,value;
           
           do{
           
               cout<<"\nChoose an operation to perform on the stack- Enter 0 to exit:"<<endl;
               cout<<"1. Push"<<endl;
               cout<<"2. Pop"<<endl;
               cout<<"3. isEmpty()"<<endl;
               cout<<"4. isFull()"<<endl;
               cout<<"5. peek()"<<endl;
               cout<<"6. count()"<<endl;
               cout<<"7. change()"<<endl;
               cout<<"8. display()"<<endl;
               cout<<"9. Clear Screen"<<endl<<endl;
               
               cin>>option;
               
               switch(option){
                   case 0:
                       break;
                   case 1:
                       cout<<"Enter a value to push in the stack: ";
                       cin>>value;
                       stk.push(value);
                       break;
                   case 2:
                       cout<<"Pop a value off the stack"<<endl;
                       stk.pop();
                       break;
                   case 3:
                       cout<<"Check if the stack is empty"<<endl;
                       if(stk.isEmpty()){
                           cout<<"Stack is empty"<<endl;
                       }else{
                           cout<<"Stack is not empty"<<endl;
                       }
                       break;	
                   case 4:
                       cout<<"Check if the stack is full"<<endl;
                       if(stk.isFull()){
                           cout<<"Stack is full"<<endl;
                       }else{
                           cout<<"Stack is not full"<<endl;
                       }
                       break;
                   case 5:
                       cout<<"Enter the position you want to peek: ";
                       cin>>position;
                       cout<<"The value in postion"<<position<<" is "<<stk.peek(position)<<endl;
                               break;
                           case 6:
                               cout<<"The number of values in the stack is "<<stk.count()<<endl;
                               break;
                           case 7:
                               cout<<"Enter the position of the item you want to change: ";
                               cin>>position;
                               cout<<"\nEnter the new value: ";
                               cin>>value;
                               stk.change(position,value);
                               break;
                           case 8:
                               cout<<"View the stack"<<endl;
                               stk.display();
                               break;
                               system("cls");
                           case 9:
                               break;
                        default:
                               cout<<"Enter options 1 - 9"<<endl;
                       }
                       
                       
                }while(option !=0);
                
                
                return 0;
            }    
}
